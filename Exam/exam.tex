function findNextId (){
  let lastId = 0;
  for (i = 0; i < data.todos.length; i++) {
      if (data.todos[i].id > lastId) {
          lastId = data.todos[i].id;
      }
  }
  lastId += 1;
  return lastId;
}
\end{lstlisting}

function filterTodos(mok, userAuth) {

      if (mok[i].userAuth == userAuth) {
          retArray.push(mok[i])
      }
  }
  return retArray
}
